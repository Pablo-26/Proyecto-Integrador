create table Hogar
(
    idHogar              varchar(250) not null,
    idSerDucha           int          null,
    idTenVivienda        int          null,
    idElBasura           int          null,
    idObAgua             int          null,
    idInSanitaria        int          null,
    idSeHigienico        int          null,
    idMaCocinar          int          null,
    idNoTieServicio      int          null,
    idVivienda           varchar(250) null,
    idMaterialTecho      int          null,
    idMaterialPared      int          null,
    idMaterialPiso       int          null,
    numeroDormitorios    int          null,
    cuartoCocina         int          null,
    numeroCuartos        int          null,
    numeroCuartosNegocio int          null,
    tieneMedidor         int          null,
    valorMensualArriendo int          null,
    parentescoPropVivi   int          null,
    pagoArriendoConLuz   int          null,
    pagoArriendoConAgua  int          null,
    juntaAgua            int          null,
    primary key (idHogar),
    constraint hogar_ibfk_1
        foreign key (idVivienda) references Vivienda (idVivienda),
    constraint hogar_ibfk_10
        foreign key (idMaterialTecho) references MaterialTecho (idMaterialTecho),
    constraint hogar_ibfk_11
        foreign key (idMaterialPiso) references MaterialPiso (idMaterialPiso),
    constraint hogar_ibfk_12
        foreign key (idMaterialPared) references MaterialPared (idMaterialPared),
    constraint hogar_ibfk_2
        foreign key (idTenVivienda) references tenenciaVivienda (idTenVivienda),
    constraint hogar_ibfk_3
        foreign key (idSerDucha) references servicioDucha (idSerDucha),
    constraint hogar_ibfk_4
        foreign key (idElBasura) references eliminarBasura (idElBasura),
    constraint hogar_ibfk_5
        foreign key (idObAgua) references obtencionAgua (idObAgua),
    constraint hogar_ibfk_6
        foreign key (idInSanitaria) references instalacionSanitaria (idInSanitaria),
    constraint hogar_ibfk_7
        foreign key (idSeHigienico) references servicioHigienico (idSeHigienico),
    constraint hogar_ibfk_8
        foreign key (idMaCocinar) references materialCocinar (idMaCocinar),
    constraint hogar_ibfk_9
        foreign key (idNoTieServicio) references noTieneServicio (idNoTieServicio)
);

create table Persona
(
    idPersona                      int auto_increment
        primary key,
    edad                           int          not null,
    cSexo_idcSexo                  int          not null,
    cEstadoCivil_idcEstadoCivil    int          not null,
    cNivelInstruccion_idcNivelInst int          not null,
    Hogar_idHogar                  varchar(100) not null,
    constraint fk_Persona_Hogar1
        foreign key (Hogar_idHogar) references Hogar (idHogar),
    constraint fk_Persona_cEstadoCivil1
        foreign key (cEstadoCivil_idcEstadoCivil) references cEstadoCivil (idcEstadoCivil),
    constraint fk_Persona_cNivelInstruccion1
        foreign key (cNivelInstruccion_idcNivelInst) references cNivelInstruccion (idcNivelInst),
    constraint fk_Persona_cSexo1
        foreign key (cSexo_idcSexo) references cSexo (idcSexo)

create index fk_Persona_Hogar1_idx
